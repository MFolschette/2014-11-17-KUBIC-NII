\begin{frame}[c]
\frametitle{Application to regulation networks}
\framesubtitle{Model from \tcite{\citeffl}}

\tval{Regulation networks} = set of species regulated by other species
\begin{fleches}
  \item The regulating species are not consumed
  \item Negative regulations \f Not always a steady state
\end{fleches}

\bigskip

\begin{center}
\scalebox{1.5}{
\begin{tikzpicture}[adn]
  \node[inner sep=0] (A) at (2,0.75) {$A$};
  \node[inner sep=0, below right of=A] (B) {$B$};
  \node[inner sep=0, below left of=A] (C) {$C$};
  \path
    (A) edge[act] (B)
    (A) edge[act] (C)
    (B) edge[inh] (C);
  
  % Knockout 1
  \uncover<3-5>{\eclair{A.east}{.5}{0}}
  \node<3-5>[ko] at (A) {$A$};
  \node<4-5>[ko] at (B) {$B$};
  \node<4-5>[ko] at (C) {$C$};
  
  % Knockout 2
  \uncover<6->{\eclair{B.north east}{.5}{40}}
  \node<6->[ko] at (B) {$B$};
  \node<7->[kobis] at (C) {$C$};
  \node<7->[pattern color=red!70, pattern=north east lines] at (C) {};
\end{tikzpicture}}
\end{center}

\bigskip

\uncover<2->{
\tval{New} notion of \tval{impact degree}
\begin{itemize}
  \item<5-> Number of species that are completely turned off
    \uncover<5->{\qf For $A$: \textcolor{red}{3}}
  \item<8-> Number of species whose behavior is modified
    \uncover<8->{\qf For $B$: \textcolor{red}{1} + \textcolor{notsodarkyellow}{1} = \textcolor{orange}{2}}
\end{itemize}
\uncover<9->{\f Requires a more precise study of the behavior}
}
